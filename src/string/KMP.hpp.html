KMP 法 (Knuth–Morris–Pratt algorithm) (src/string/KMP.hpp)

使い方

  • vector<ll> KMP(string s)：$\text{kmp}[i] := \max\{ l ≤ i \mid s[:l] = s[(i+1)-l:i+1] \}$ で定義される配列 $\text{kmp}$ を求める

$O(n)$ 時間

// kmp[i] := max{ l ≤ i | s[:l] == s[(i+1)-l:i+1] }
// abacaba -> 0010123
auto KMP(string s) {
   V<ll> p(sz(s));
   rep(i, 1, sz(s)) {
      ll g = p[i - 1];
      while(g && s[i] != s[g]) g = p[g - 1];
      p[i] = g + (s[i] == s[g]);
   }
   return p;
}
